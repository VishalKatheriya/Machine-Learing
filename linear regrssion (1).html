#2. wap to find linear regrssion of two points using gradient descent
In [19]:
a=0
b=0
lr=0.00001
iteration=1000
error=[]
for itr in range(iteration):
    errorcost=0
    costa=0
    costb=0
    for i in range(len(x)):
        ypred=a*x[i]+b
        errorcost=(y[i]-ypred)**2
        for j in range(len(x)):
            pda=(-2*x[j]*(y[j]-(a*x[j]+b)))
            pdb=-2*(y[j]-(a*x[j]+b))
            costa+=pda
            costb+=pdb
        a-=lr*costa
        b-=lr*costb
        error.append(errorcost)
