def find_cluster(X,n_cluster,r_seed=2):
    #generate random state
    random_state=np.random.RandomState(r_seed)
    print('random state:',random_state)
    i=random_state.permutation(x.shape[0])[:n_cluster]
    print('permutations: ',i)
    centers=x[i]
    print('initial centers: ',centers)
    print('converging centers:')
    while True:
        label=pairwise_distances_argmin(x,centers)
        print('calculated distances: ',label)
        new_center=np.array([x[label==i].mean(0) for i in range(n_cluster)])
        print('updated centers: ',new_center)
        if np.all(centers==new_center):          #test whether all array elements along a given axis evaluate to True
            break
        centers=new_center
        print(centers)
        print()
    return label,centers
