#now for 2 to 7 degree
for i in range(2,7):
    print("degree:",i)
    poly=PolynomialFeatures(degree=i)
    poly_feature=poly.fit_transform(x.reshape(-1,1))
    poly_reg_model=LinearRegression()
    poly_reg_model.fit(poly_feature,y)
    y_pred=poly_reg_model.predict(poly_feature)
    plt.figure(figsize=(8,6))
    plt.title("polynomial regression:",size=10)
    plt.scatter(x,y)
    plt.plot(x,y_pred,c='black')
    plt.show()
    print("MSE:",np.sqrt(mean_squared_error(y,y_pred)))
